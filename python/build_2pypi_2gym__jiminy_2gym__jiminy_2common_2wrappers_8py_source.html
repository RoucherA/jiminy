26 def flatten_observation(space, x=None):
27  # Note that it does not preserve dtype
28  def _flatten_bounds(space, bounds_type):
29  if isinstance(space, spaces.Box):
30  if bounds_type == 'high':
31  return np.asarray(space.high).flatten()
32  else:
33  return np.asarray(space.low).flatten()
34  elif isinstance(space, spaces.Discrete):
35  if bounds_type == 'high':
36  return np.one(space.n)
37  else:
38  return np.zeros(space.n)
39  elif isinstance(space, spaces.Tuple):
40  return np.concatenate([_flatten_bounds(s, bounds_type)
41  for s in space.spaces])
42  elif isinstance(space, spaces.Dict):
43  return np.concatenate([_flatten_bounds(s, bounds_type)
44  for s in space.spaces.values()])
45  elif isinstance(space, spaces.MultiBinary):
46  if bounds_type == 'high':
47  return np.one(space.n)
48  else:
49  return np.zeros(space.n)
50  elif isinstance(space, spaces.MultiDiscrete):
51  if bounds_type == 'high':
52  return np.one(reduce(__mul__, space.nvec))
53  else:
54  return np.zeros(reduce(__mul__, space.nvec))
55  else:
56  raise NotImplementedError
57  if x is None:
58  return spaces.Box(low=_flatten_bounds(space, 'low'),
59  high=_flatten_bounds(space, 'high'),
60  dtype=np.float64)
61  else:
62  return spaces.flatten(space, x)
